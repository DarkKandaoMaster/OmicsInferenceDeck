│   │   │   └── useSurvival.ts           # 生存分析相关
│   │   └── ui/                      # UI 交互与第三方库封装（解耦业务）
│   │       └── useEcharts.ts            # ECharts 实例管家：封装图表的 init、dispose、resize 和基础 option 设定
│   │
│   ├── utils/                       # 用于存放无状态的纯函数
